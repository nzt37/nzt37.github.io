  • P1069 细胞分裂
    在考虑一个数是否可以整除另一个数的时候,观察前一个数是否包含后一个数的所有质因数,
    在边输入边判断时,一定要考虑是否将上一次判断留下来的数据清零,(在这里卡了好长时间)
#include <bits/stdc++.h>
using namespace std;
const int maxn = 3000;
int minn = 214783647,now;
int p[maxn],cell[maxn],prime[maxn],n,m1,m2,cnt,count;
bool vis = true,flag = false;
bool pr(int k)  //判断是否为素数，不予以注释
{
    int m;
    for(m=2;m<=floor(sqrt(k));m++)
    if(k%m==0)return false;
    return true;
}
int main()
{
	cin>>n;
	cin>>m1>>m2;
	for(int i=2;i<=maxn;i++)  //把30000以内的质因数枚举出来
    if(pr(i))
    {
    	cnt++;
        prime[cnt]=i;
    }
	int i = 1;
	while(m1!=1)
	{
		if(m1%prime[i] == 0)
		{
			while(m1%prime[i] == 0)
			{
				p[i] += m2;
				m1/=prime[i];
			}
			
		}
		i++;
	}
	for(int j = 1 ; j <= n ; j++)
	{
		int a;
		cin>>a;
		 memset(cell,0,sizeof(cell)); 
		 now = 0,vis = true;
		int i = 1;
		while(i <= cnt)
		{
			if(a%prime[i]==0)
			{
				while(a%prime[i] == 0)
				{
					cell[i]++;
					a /= prime[i];
				}
			}
			i++;
		}
		for(int i = 1 ; i <= cnt ; i++)
		{
			if(p[i]!=0&&cell[i]==0) vis = false;
			if(p[i]!=0&&cell[i]!=0)
			{
				if(p[i]%cell[i]==0){
					now = max(now,p[i]/cell[i]);
				}
				else
				now = max(now,p[i]/cell[i]+1);
			}
		}
		if(vis)
		{
			flag = true;
			if(now<minn) minn = now;
		}
	}
	if(flag)cout<<minn<<endl;  //如果可行
    else cout<<-1<<endl;
	return 0;
 }
